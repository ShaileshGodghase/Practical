#include <bits/stdc++.h>
using namespace std;

int isPrime(int x)
{
   int s = sqrt(x);

   for (int i = 2; i <= s; i++)

      if (x % i == 0)
         return 0;

   return 1;
}

vector<int> primeFactors(int n) 
{  vector<int> factors;
    while (n%2 == 0) 
    { 
        factors.push_back(2);
        n = n/2; 
    } 
     
    for (int i = 3; i <= sqrt(n); i = i+2) 
    {  
        while (n%i == 0) 
        { 
            factors.push_back(i); 
            n = n/i; 
        } 
    } 
     
    if (n > 2) 
        factors.push_back(n);

   return factors; 
} 

int sumOfDigits(int n)
{
   int sum = 0, x = n;

   while(x > 0)
   {
      int last = x % 10;
      sum = sum + last;
      x = x / 10;
   }
   return sum;
}

int main()
{
   int t; cin>>t;

   while(t--)
   {  
      int n; cin>>n;

      for(int i=n;i<1e9;i++)
      {
         vector<int> factors = primeFactors(i);
         int sum = 0;
         for(auto it: factors)
         {
            sum = sum + sumOfDigits(it);
         }
         int n_sum = sumOfDigits(i);
         if(n_sum == sum)
         {
            cout<<i<<endl;
            break;
         }
      }
   }
   return 0;
}
