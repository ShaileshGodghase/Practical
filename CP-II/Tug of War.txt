#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t; cin>>t;

   while(t--)
   {
      int n; cin>>n;

      vector<int> vect(n);

      for(int i=0;i<n;i++)
      {
         cin>>vect[i];
      }
      sort(vect.begin(),vect.end());

      vector<int> t1,t2;
      int c1 = vect[n-1], c2 = vect[n-2];

      t1.push_back(vect[n-1]);
      t2.push_back(vect[n-2]);

      for(int i=n-3;i>=0;i--)
      {
         if(c2 < c1)
         {
            t2.push_back(vect[i]);
            c2 = c2 + vect[i];
         }
         else
         {
            t1.push_back(vect[i]);
            c1 = c1 + vect[i];
         }
      }
      cout<<min(c1,c2)<<" "<<max(c1,c2)<<endl;
   }
   return 0;
}
