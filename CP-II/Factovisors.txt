#include<bits/stdc++.h>
using namespace std;


bool CheckIfPossible(int n, int prime, int count)
{
    long primeValue = prime;
    
    while (n / primeValue && count > 0)
    {
        count -= n / primeValue;
        primeValue *= prime;
    }
    
    return count <= 0;
}


int main()
{
    int n, d, count, temp;
    bool isDivisor;
    
    while (cin>>n>>d)
    {
        temp = d;
        isDivisor = true;
        
        for (int i = 2; i * i <= d && isDivisor; ++i)
        {
            count = 0;
            
            while (d % i == 0)
            {
                d /= i;
                ++count;
            }
            
            if (count > 0)
                isDivisor = CheckIfPossible(n, i, count);
        }
        
        if (d > 1 && isDivisor)
        {
            isDivisor = CheckIfPossible(n, d, 1);
        }
        
        string ans = "";
        if(isDivisor) ans = "divides";
        else ans = "does not divide";

        cout<<temp<<" "<<ans<<" "<<n<<"!"<<endl;
    }
}
