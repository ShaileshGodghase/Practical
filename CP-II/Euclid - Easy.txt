#include<bits/stdc++.h>
using namespace std;

bool check(double n){
    if(n < 0) n = n * -1;
   if(ceil(n) == floor(n))
        return true;
    
    return false;
 }

int main()
{
    int a,b; cin>>a>>b;
    int g = __gcd(a,b);
    
    vector<pair<int,int>> possible_pair;
    
    int large = max(a,b);
    for(int i=0;i<=large;i++)
    {
        double x = (double)(g - (b * i)) / a;
        //cout<<"x = "<<x<<endl;
        if(check(x))
        {
            possible_pair.push_back({x,i});
        }
    }
    
    int ans1 = min(possible_pair[0].first,possible_pair[0].second);
    int ans2 = max(possible_pair[0].first,possible_pair[0].second);
    cout<<ans1<<" "<<ans2<<" "<<g<<endl;
    return 0;
}
