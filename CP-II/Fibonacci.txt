#include<bits/stdc++.h>
using namespace std;

int fibo(int n) {
   int fibo[n+2];
   fibo[0] = 0;
   fibo[1] = 1;
   for (int i = 2; i <= n; i++) {
      fibo[i] = fibo[i-1] + fibo[i-2];
   }
   return fibo[n];
}

int fibonacci(int n)
{
    if (n <= 1)
        return n;
    return fibonacci(n-1) + fibonacci(n-2);
}

int main()
{
   cout<<fibonacci(3);
   return 0;
}
